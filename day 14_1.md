    result = []

# 구구단 함수_2
def GuGu2(n):
    result = []
    i = 1
    while i < 10:
        result.append(n * i)
        i = i + 1
    return result

print(GuGu2(2))
# [2, 4, 6, 8, 10, 12, 14, 16, 18]

def getTotalPage(m, n):
    if m % n == 0:
        return m //n
    else:
        return m // n + 1

print(getTotalPage(5, 10))
